class Calculate {
    constructor(price) {
        this.price = price
    }
    getPrice() {
        return `The User price is: ${this.price * 100}`
    }
}

const person1 = new Calculate(1039)
const person2 = new Calculate(200)

console.log(person2.getPrice())
